Pailwater Darts Tournament

    p1score=0
    p2score=0
    total1=0
    total2=0
    
    print("What was player ones score?")
    for counter in range(3):
    	p1score=int(input())
    	while p1score < 0 or p1score >180:
    		print("Invalid score! Enter again please:")
    		p1score=int(input())
    	total1=total1+p1score
    
    print("What was player twos score?")
    for counter in range(3):
    	p2score=int(input())
    	while p2score < 0 or p2score >180:
    		print("Invalid score! Enter again please:")
    		p2score=int(input())
    	total2=total2+p2score
    
    
    print("Player ones score was",total1)
    print("Player twos score was",total2)
    
    if total1>total2:
      print("Player one wins!")
    
    elif total2>total1:
      print("Player two wins!")
    
    elif total1==total2:
      print("It was a draw")
