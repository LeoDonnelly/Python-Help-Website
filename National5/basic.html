Calculating the atomic weight of hydrocarbons

    carbonattom = 0
    hydroatom = 0
    mass = 0
    
    carbonattom = int(input("Enter the number of carbon atoms? \n"))
    
    hydroatom = (carbonattom*2)+2
    mass = carbonattom*12+hydroatom
    
    print("The atomic mass of C" + str(carbonattom) + "H" + str(hydroatom), "is", str(mass))
